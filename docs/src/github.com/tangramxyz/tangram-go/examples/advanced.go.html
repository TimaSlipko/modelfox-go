    11  func main() {
    12  	// If you are running the Tangram app on your own server you can pass the URL to it with the TANGRAM_URL environment variable.
    13  	tangramURL, present := os.LookupEnv("TANGRAM_URL")
    14  	if !present {
    15  		tangramURL = "https://app.tangram.xyz"
    16  	}
    17  
    18  	// Load the model from the path.
    19  	options := tangram.LoadModelOptions{
    20  		TangramURL: tangramURL,
    21  	}
    22  	model, err := tangram.LoadModelFromPath("examples/heart_disease.tangram", &options)
    23  	if err != nil {
    24  		log.Fatal(err)
    25  	}
    26  	// Destroy the model when it is no longer needed to free up memory.
    27  	defer model.Destroy()
    28  
    29  	// Create an example input matching the schema of the CSV file the model was trained on. Here the data is just hard-coded, but in your application you will probably get this from a database or user input.
    30  	input := tangram.PredictInput{
    31  		"age":                                  63,
    32  		"gender":                               "male",
    33  		"chest_pain":                           "typical angina",
    34  		"resting_blood_pressure":               145,
    35  		"cholesterol":                          233,
    36  		"fasting_blood_sugar_greater_than_120": "true",
    37  		"resting_ecg_result":                   "probable or definite left ventricular hypertrophy",
    38  		"exercise_max_heart_rate":              150,
    39  		"exercise_induced_angina":              "no",
    40  		"exercise_st_depression":               2.3,
    41  		"exercise_st_slope":                    "downsloping",
    42  		"fluoroscopy_vessels_colored":          "0",
    43  		"thallium_stress_test":                 "fixed defect",
    44  	}
    45  
    46  	// Make the prediction using a custom threshold chosen on the "Tuning" page of the Tangram app.
    47  	predictOptions := tangram.PredictOptions{
    48  		Threshold:                   0.5,
    49  		ComputeFeatureContributions: true,
    50  	}
    51  	output := model.PredictOne(input, &predictOptions).(tangram.BinaryClassificationPredictOutput)
    52  
    53  	// Print the output.
    54  	fmt.Println("Output:", output)
    55  
    56  	// Log the prediction.
    57  	err = model.LogPrediction(tangram.LogPredictionArgs{
    58  		Identifier: "71762b29-2296-4bf9-a1d4-59144d74c9d9",
    59  		Input:      input,
    60  		Options:    predictOptions,
    61  		Output:     output,
    62  	})
    63  	if err != nil {
    64  		log.Fatal(err)
    65  	}
    66  
    67  	// Later on, if we get an official diagnosis for the patient, log the true value. Make sure to match the `identifier`.
    68  	err = model.LogTrueValue(tangram.LogTrueValueArgs{
    69  		Identifier: "71762b29-2296-4bf9-a1d4-59144d74c9d9",
    70  		TrueValue:  "Positive",
    71  	})
    72  	if err != nil {
    73  		log.Fatal(err)
    74  	}
    75  }
