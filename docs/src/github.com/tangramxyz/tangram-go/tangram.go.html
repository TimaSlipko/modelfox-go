   246  	var cModel *C.tangram_model
   247  	cPath := C.CString(path)
   248  	err := C.tangram_model_from_path(cPath, &cModel)
   249  	if err != nil {
   250  		var sv C.tangram_string_view

   252  		C.tangram_error_get_message(err, &sv)
   253  		errs := C.GoStringN(sv.ptr, C.int(sv.len))
   254  		return nil, errors.New(errs)
   255  	}
   256  	queue := []event{}
   257  	model := Model{
   258  		cModel,
   259  		options,
   260  		queue,
   261  	}
   262  	return &model, nil
   263  }

   310  func newPredictInput(input PredictInput) *C.tangram_predict_input {
   311  	var cInput *C.tangram_predict_input
   312  	C.tangram_predict_input_new(&cInput)
   313  	var cKey *C.char
   314  	defer C.free(unsafe.Pointer(cKey))
   315  	for key, value := range input {
   316  		switch value.(type) {
   317  		case string:
   318  			cKey = C.CString(key)
   319  			cVal := C.CString(value.(string))
   320  			err := C.tangram_predict_input_set_value_string(cInput, cKey, cVal)
   321  			if err != nil {
   322  				logTangramError(err)
   323  			}
   324  		case float64:
   325  			cKey = C.CString(key)
   326  			cVal := C.double(float64(value.(float64)))
   327  			err := C.tangram_predict_input_set_value_number(cInput, cKey, cVal)
   328  			if err != nil {
   329  				logTangramError(err)
   330  			}
   331  		case int:
   332  			cKey = C.CString(key)
   333  			cVal := C.double(float64(value.(int)))
   334  			err := C.tangram_predict_input_set_value_number(cInput, cKey, cVal)
   335  			if err != nil {
   336  				logTangramError(err)
   337  			}
   338  		case bool:
   339  			cKey = C.CString(key)
   340  			cVal := C.CString(strconv.FormatBool(value.(bool)))
   341  			err := C.tangram_predict_input_set_value_string(cInput, cKey, cVal)
   342  			if err != nil {
   343  				logTangramError(err)
   344  			}
   345  		}
   346  	}
   347  	return cInput
   348  }

   374  // Make a prediction with multiple inputs.
   375  func (m Model) Predict(input []PredictInput, options *PredictOptions) []PredictOutput {
   376  	var cOutputVec *C.tangram_predict_output_vec
   377  	cInputVec := newPredictInputVec(input)
   378  	cOptions := newPredictOptions(options)
   379  	defer C.tangram_predict_options_delete(cOptions)
   380  	defer C.tangram_predict_input_vec_delete(cInputVec)
   381  	err := C.tangram_model_predict(m.modelPtr, cInputVec, cOptions, &cOutputVec)
   382  	if err != nil {
   383  		logTangramError(err)
   384  	}
   385  	defer C.tangram_predict_output_vec_delete(cOutputVec)
   386  
   387  	outputVec := make([]PredictOutput, len(input))
   388  	var cTaskType C.tangram_task
   389  	C.tangram_model_get_task(m.modelPtr, &cTaskType)
   390  	for i := 0; i < len(input); i++ {
   391  		var cOutput *C.tangram_predict_output
   392  		C.tangram_predict_output_vec_get_at_index(cOutputVec, C.size_t(i), &cOutput)
   393  		outputVec[i] = makePredictOutputFromTangramPredictOutput(cTaskType, cOutput)
   394  	}
   395  	return outputVec
   396  }

   409  	}
   410  	return nil

   453  // A helper function to extract a MulticlassClassificationPredictOutput from a *C.tangram_predict_output.
   454  func makeMulticlassClassificationPredictOutputFromTangramPredictOutput(output *C.tangram_predict_output) MulticlassClassificationPredictOutput {
   455  	var cOutput *C.tangram_multiclass_classification_predict_output
   456  	var cProbability C.float
   457  	var sv C.tangram_string_view
   458  	C.tangram_predict_output_as_multiclass_classification(output, &cOutput)
   459  	C.tangram_multiclass_classification_predict_output_get_probability(cOutput, &cProbability)
   460  	C.tangram_multiclass_classification_predict_output_get_class_name(cOutput, &sv)
   461  	predictedClassName := C.GoStringN(sv.ptr, C.int(sv.len))
   462  	var cClassProbability C.float
   463  	var cProbabilitiesIter *C.tangram_multiclass_classification_predict_output_probabilities_iter
   464  	C.tangram_multiclass_classification_predict_output_get_probabilities_iter(cOutput, &cProbabilitiesIter)
   465  	defer C.tangram_multiclass_classification_predict_output_probabilities_iter_delete(cProbabilitiesIter)
   466  	probabilities := make(map[string]float32)
   467  	for C.tangram_multiclass_classification_predict_output_probabilities_iter_next(cProbabilitiesIter, &sv, &cClassProbability) {
   468  		className := C.GoStringN(sv.ptr, C.int(sv.len))
   469  		probabilities[className] = float32(cClassProbability)
   470  	}
   471  	var cFeatureContributionsIter *C.tangram_multiclass_classification_predict_output_feature_contributions_iter
   472  	C.tangram_multiclass_classification_predict_output_get_feature_contributions_iter(cOutput, &cFeatureContributionsIter)
   473  	defer C.tangram_multiclass_classification_predict_output_feature_contributions_iter_delete(cFeatureContributionsIter)
   474  	featureContributions := make(map[string]FeatureContributions)
   475  	if cFeatureContributionsIter != nil {
   476  		var cFeatureContributions *C.tangram_feature_contributions

   687  func (m Model) logEvents(events []event) error {
   688  	body, err := json.Marshal(events)
   689  	if err != nil {
   690  		return err
   691  	}
   692  	req, err := http.NewRequest(
   693  		"POST",
   694  		m.options.TangramURL+"/track",
   695  		bytes.NewReader(body),
   696  	)
   697  	if err != nil {
   698  		return err
   699  	}
   700  	req.Header.Set("Content-Type", "application/json")
   701  	res, err := http.DefaultClient.Do(req)
   702  	if err != nil {
   703  		return err
   704  	}
   705  	defer res.Body.Close()
   706  	if res.StatusCode < 200 || res.StatusCode > 299 {
   707  		body, err := ioutil.ReadAll(res.Body)
   708  		if err != nil {
   709  			return err
   710  		}
   711  		return errors.New(string(body))
   712  	}
   713  	return nil
   714  }
